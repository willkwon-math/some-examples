def fib(n):
    a,b = 1,1
    if n==1 or n==2:
        return 1
    for i in range(1,n):
        a,b = b, a+b
    return a

result = str(" ")

for i in range(1,20):
	result = str(result) + str(i) +str("&")+ str(fib(i)) + str("\\tabularnewline")  
\end{pycode}
